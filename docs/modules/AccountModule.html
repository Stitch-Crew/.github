import {
  MiddlewareConsumer,
  Module,
  NestModule,
  RequestMethod,
  forwardRef,
} from '@nestjs/common';
import { AccountService } from './account.service';
import { AccountController } from './account.controller';
import { AccountRepository } from 'src/app/shared/repositories/account';
import { TokenModule } from '../token/token.module';
import { AccountSchema } from 'src/app/shared/schemas/account';
import { MongooseModule } from '@nestjs/mongoose';
import { LoggerMiddleware } from 'src/app/shared/middleware/logger.middleware';

/**
 * A module for managing user accounts.
 *
 * @module AccountModule
 */
@Module({
  imports: [
    MongooseModule.forFeature([{ name: 'Account', schema: AccountSchema }]),
    TokenModule,
  ],
  providers: [AccountService, AccountRepository],
  controllers: [AccountController],
  exports: [AccountService, AccountRepository],
})
export class AccountModule implements NestModule {
  configure(consumer: MiddlewareConsumer) {
    consumer
      .apply(LoggerMiddleware)
      .forRoutes(
        { path: '/account', method: RequestMethod.POST },
        { path: '/account/login', method: RequestMethod.POST },
        { path: '/account/refresh', method: RequestMethod.POST },
      );
  }
}
