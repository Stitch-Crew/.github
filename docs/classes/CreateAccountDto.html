import { ApiProperty } from '@nestjs/swagger';
import {
  IsString,
  IsEmail,
  MinLength,
  IsNotEmpty,
  MaxLength,
  Matches,
  IsDefined,
} from 'class-validator';
import { config } from '../../../../config/config';

const MIN_PASSWORD_LENGTH = config.minPasswordLength;
const MAX_PASSWORD_LENGTH = config.maxPasswordLength;

const checkUpperCaseString = /(?=.*?[A-Z])/;
const checkUpperCaseStringErrMsg =
  'Your password must include uppercase character';

const checkLowerCaseString = /(?=.*?[a-z])/;
const checkLowerCaseStringErrMsg =
  'Your password must include lowercase character';

const checkNumberString = /(?=.*?[0-9])/;
const checkNumberStringErrMsg = 'Your password must include numbers';

const checkSpecialString = /(?=.*?[!“#$%&‘()*+,-./:;<=>?@[\]^_`{|}~])/;
const checkSpecialStringErrMsg = 'Your password must include special character';

const checkEmptySpaceString = /^((?![\s]).)*$/;
const checkEmptySpaceStringErrMsg = 'Your password cannot have empty spaces';

/**
 * Account validation object for creation
 */
export class CreateAccountDto {
  /**
   * Email to access the platform
   */
  @IsEmail()
  @IsNotEmpty()
  @ApiProperty({
    description: 'Email to register user in our application',
    example: 'user@domain.com',
  })
  @IsDefined()
  email: string;

  /**
   * Password to access the platform
   */
  @IsString()
  @MinLength(MIN_PASSWORD_LENGTH)
  @MaxLength(MAX_PASSWORD_LENGTH)
  @IsNotEmpty()
  @ApiProperty({
    description: 'Password to register the user in our application',
    minimum: 8,
    maximum: 128,
    example: '123Mudar!',
  })
  @IsDefined()
  @Matches(checkUpperCaseString, { message: checkUpperCaseStringErrMsg })
  @Matches(checkLowerCaseString, { message: checkLowerCaseStringErrMsg })
  @Matches(checkNumberString, { message: checkNumberStringErrMsg })
  @Matches(checkSpecialString, { message: checkSpecialStringErrMsg })
  @Matches(checkEmptySpaceString, { message: checkEmptySpaceStringErrMsg })
  password: string;
}
