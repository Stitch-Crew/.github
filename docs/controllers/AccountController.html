import {
  Controller,
  Post,
  Body,
  Request,
  UnauthorizedException,
} from '@nestjs/common';
import { AccountService } from './account.service';
import {
  AuthenticationResponseDto,
  CreateAccountDto,
  RefreshTokenDto,
} from '../../shared/dto/account/';
import { ApiTags } from '@nestjs/swagger';
import { TokensService } from '../token/token.service';
import { Account } from '../../shared/interfaces/account';
import { LoginDto } from '../../shared/dto/account/';
import { ResponseInterface } from '../../core/interfaces/response.interface';
import { ApiEndpoint } from 'src/app/core/decorators/api-endpoint';

@ApiTags('account')
@Controller('account')
export class AccountController {
  private readonly expiration: number;

  public constructor(
    private accountService: AccountService,
    private tokensService: TokensService,
  ) {
    this.expiration = 60 * 60 * 24 * 30;
  }

  @ApiEndpoint({
    success: {
      status: 201,
      dto: AuthenticationResponseDto,
    },
    validationMessages: [
      'Invalid token id',
      'Invalid account id',
      'Email already in use',
    ],
  })
  @Post('/')
  public async register(
    @Body() body: CreateAccountDto,
  ): Promise<ResponseInterface<AuthenticationResponseDto>> {
    const user = await this.accountService.createUserFromRequest(body);
    const token = await this.tokensService.generateAccessToken(user);
    const refresh = await this.tokensService.generateRefreshToken(
      user,
      this.expiration,
    );

    const payload = await this.buildResponsePayload(user, token, refresh);
    return {
      status: 'success',
      data: payload,
    };
  }

  @ApiEndpoint({
    success: {
      status: 201,
      dto: AuthenticationResponseDto,
    },
    error: {
      status: 401,
      message: {
        example: 'The login is invalid',
      },
      error: {
        example: 'Unauthorized',
      },
    },
  })
  @Post('/login')
  public async login(
    @Body() body: LoginDto,
  ): Promise<ResponseInterface<AuthenticationResponseDto>> {
    const { email, password } = body;

    const account = await this.accountService.findForEmail(email);
    const valid = account
      ? await this.accountService.validateCredentials(account, password)
      : false;

    if (!valid) {
      throw new UnauthorizedException('The login is invalid');
    }

    const payload = await this.generateToken(account);

    return {
      status: 'success',
      data: payload,
    };
  }

  @ApiEndpoint({
    auth: true,
    success: {
      status: 201,
      dto: AuthenticationResponseDto,
    },
    validationMessages: ['The refresh token is required'],
    error: {
      status: 401,
      message: {
        examples: ['Refresh token expired', 'Refresh token malformed'],
      },
      error: {
        example: 'Unauthorized',
      },
    },
  })
  @Post('/refresh')
  public async refresh(
    @Body() body: RefreshTokenDto,
  ): Promise<ResponseInterface<AuthenticationResponseDto>> {
    const { account, token } =
      await this.tokensService.createAccessTokenFromRefreshToken(
        body.refresh_token,
      );

    const payload = await this.buildResponsePayload(account, token);

    return {
      status: 'success',
      data: payload,
    };
  }

  private async generateToken(account: Account) {
    const token = await this.tokensService.generateAccessToken(account);
    const refresh = await this.tokensService.generateRefreshToken(
      account,
      this.expiration,
    );

    return await this.buildResponsePayload(account, token, refresh);
  }

  private async buildResponsePayload(
    account: Account,
    accessToken: string,
    refreshToken?: string,
  ): Promise<AuthenticationResponseDto> {

    return {
      account: {
        id: account.id,
        email: account.email,
      },
      payload: {
        type: 'bearer',
        token: accessToken,
        ...(refreshToken ? { refresh_token: refreshToken } : {}),
        ttl: this.expiration,
      }
    };
  }
}
