import { HttpService } from '@nestjs/axios';
import { Injectable, UnprocessableEntityException } from '@nestjs/common';
import { AxiosResponse } from '@nestjs/terminus/dist/health-indicator/http/axios.interfaces';
import { lastValueFrom } from 'rxjs';
import { config } from '../../../config/config';
import { DecryptedResponse, EncryptedResponse, HandshakeResponse } from 'src/app/shared/interfaces/encryption';
import { HeimdallService } from 'src/app/core/heimdall/heimdall.service';


@Injectable()
export class EncryptionService {

  constructor(private readonly httpService: HttpService, private readonly haimdallService: HeimdallService) { }

  private async serverGenerateKeys(token: string){
    const { privateKey, publicKey } = await this.haimdallService.generate()

    process.env[`${token}_api_secret`] = privateKey;
    process.env[`${token}_api_public`] = publicKey;
  }

  async encrypt(
    token: string,
    data: object
  ): Promise<DecryptedResponse>{
      try {
        
        const handshakeResponse = await this.performHandshake(token);

        const armoredPrivateKey = process.env[`${token}_api_secret`];
        const armoredPublicKey = handshakeResponse.data.publicKey;
        
        const encryptedPayload = await this.haimdallService.encrypt(data, armoredPrivateKey, armoredPublicKey)

        const certificate$ = this.httpService.post(
          `${config.saltEncrypt}/api/encrypt`,
          {
            encrypted: encryptedPayload
          },
          {
            headers: {
              'Content-Type': 'application/json',
              'token': token
            },
          },
        );
        const encryptedResponse = await lastValueFrom<AxiosResponse<EncryptedResponse>>(certificate$);
        const { decrypted, signatures } = await this.haimdallService.decrypt(encryptedResponse.data.encrypted, armoredPrivateKey, armoredPublicKey)
        const signature = await signatures[0].verified;
        if(!signature){
          throw new UnprocessableEntityException('Signature not valid into data encryption between api and service')
        }
        return { data: decrypted, signature: true}
        
      } catch (error) {
        throw error;
      }

  }

  async decrypt(
    token: string,
    data: string
  ): Promise<DecryptedResponse> {
    try {

      const handshakeResponse = await this.performHandshake(token);

      const armoredPrivateKey = process.env[`${token}_api_secret`];
      const armoredPublicKey = handshakeResponse.data.publicKey;

      const encryptedPayload = await this.haimdallService.encrypt(data, armoredPrivateKey, armoredPublicKey)

      const certificate$ = this.httpService.post(
        `${config.saltEncrypt}/api/decrypt`,
        {
          encrypted: encryptedPayload
        },
        {
          headers: {
            'Content-Type': 'application/json',
            'token': token
          },
        },
      );
      const encryptedResponse = await lastValueFrom<AxiosResponse<EncryptedResponse>>(certificate$);
      const { decrypted, signatures } = await this.haimdallService.decrypt(encryptedResponse.data.encrypted, armoredPrivateKey, armoredPublicKey)
      const signature = await signatures[0].verified;
      if (!signature) {
        throw new UnprocessableEntityException('Signature not valid into data encryption between api and service')
      }
      const parsedDecription = decrypted as string;
      return { data: JSON.parse(parsedDecription), signature: true }

    } catch (error) {
      throw error;
    }

  }

  private async performHandshake(token: string): Promise<AxiosResponse<HandshakeResponse>> {
    try {
      await this.serverGenerateKeys(token);

      const handshake$ = this.httpService.post(
        `${config.saltEncrypt}/api/handshake`,
        {
          apiPublicKey: process.env[`${token}_api_public`], // Send the public key of your API
          token: token, // Send the unique token
        },
        {
          headers: {
            'Content-Type': 'application/json',
          },
        },
      );

      const response = await lastValueFrom(handshake$);
      return response;
    } catch (error) {
      throw error;
    }
  }

}
